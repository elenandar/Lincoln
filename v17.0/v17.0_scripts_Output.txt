const modifier = (text) => {
    try {
        // В Phase 0 НИЧЕГО не анализируем, просто пропускаем.
        // Если text пустой/undefined — возвращаем одиночный пробел.
        return { text: (text && text !== "") ? text : " " };
    } catch (e) {
        console.log("Output modifier error:", e);
        // Fallback чтобы не вызвать ошибку сценария
        return { text: (text && text !== "") ? text : " " };
    }
};

modifier(text);
