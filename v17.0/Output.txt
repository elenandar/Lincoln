const modifier = (text) => {
  const LC = state.shared.LC;
  if (typeof LC === 'undefined') {
    return;
  }

  const L = LC.lcInit('Output');

  if (L.currentAction.type === 'story') {
    LC.Turns.inc();
  }
  
  // Примечание: state.message очищается движком после отображения.
  // Мы больше не используем consumeMsgs в Output, так как команды
  // теперь напрямую пишут в state.message.
};

modifier(text);
